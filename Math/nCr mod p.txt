ll modPow(ll a,ll n) 
{
  ll res = 1;
  while (n) 
  {
    if(n%2==0)
    {
      a=(a*a)%mod;
      n/=2;
    }
    else
    {
      res=(res*a)%mod;
      n--;
    }
  }
  return res;
}

unsigned long long modInverse(ll n)
{
    return modPow(n,mod-2);
}

ll nCr(ll n,ll r)
{
    if(n<=0 || r<0 || n<r){return 0;}
    else if(r==0){return 1;}
    
    return (fact[n] * (modInverse(fact[r]))%mod * (modInverse(fact[n-r]))%mod ) % mod;
}
