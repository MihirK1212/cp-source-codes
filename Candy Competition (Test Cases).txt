#include <iostream>
using namespace std;
int main()
{
    long long sum=0,p,t,n,k,i,par,j;
    cout<<"Enter number of test cases:\n";
    cin>>t;
    long long ans[t];
    long long a[t][1000000]; 
    for(p=0;p<t;p++)
    {
     cout<<"For test case number "<<p+1<<"\n";
     cout<<"Enter number of candies:\n";
     cin>>n;
     i=0;
     sum=0;
     k=0;
     while(sum<n)
     {
         if((sum+i+1)<=n)
         {
            a[p][i]=i+1;
            sum=sum+(i+1); 
         }
         else if((sum+i+1)>n)
         {
             k=i;
             a[p][i-1]=n+a[p][i-1]-sum;
             break;
         }
         if(sum==n)
         {
             k=i+1;
             break;
         }
         i++;
     } 
     ans[p]=k;
    
    }
    
    int c;
    for(p=0;p<t;p++)
    {
        c=1;
        cout<<"For test case number "<<p+1<<"\n";
        cout<<"Number of rankers= "<<ans[p]<<"\n";
        for(j=ans[p]-1;j>=0;j--)
        {
            cout<<"a"<<c<<"= "<<a[p][j]<<"\n";
            c++;
        }
    }
}
