#include <iostream>
#include <vector>
using namespace std;

int partition(vector<int> &arr,int lb,int ub)
    {
        int pivot,i,j,temp;
         pivot=arr[lb];
         j=lb;
         for(i=lb+1;i<=ub;i++)
         {
             if(arr[i]<=pivot)
             {
                 temp=arr[j+1];
                 arr[j+1]=arr[i];
                 arr[i]=temp;
                 j++;
             }
         }
         temp=arr[lb];
         arr[lb]=arr[j];
         arr[j]=temp;
         return j;
    }


void quicksort(vector<int> &arr,int low,int high)
{
    int m;
    if(low>=high)
    {
        return;
    }
    else
    {
        int x,random;
        random=(low+high)/2;
        x=arr[low];
        arr[low]=arr[random];
        arr[random]=x;
        
        m=partition(arr,low,high);
        
        quicksort(arr,low,m-1);
        quicksort(arr,m+1,high);
    }
}

 


int main()
{
    int n,k;
    vector<int> arr(n);
    cout<<"Enter number of elements:\n";
    cin>>n;
    cout<<"Enter elements of array\n";
    for(k=0;k<n;k++)
    {
        cin>>arr[k];
    }
    quicksort(arr,0,n-1);
    cout<<"Sorted array is:\n";
    for(k=0;k<n;k++)
    {
        cout<<arr[k]<<"\n";
    }
    
  return 0;
}
