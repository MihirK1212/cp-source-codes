
#include <bits/stdc++.h>

using namespace std;

bool check(vector<vector<int>> &grid)
{
    int count=0,i,j;
    
    for(i=0;i<3;i++)
    {
        count=0;
        for(j=0;j<=1;j++)
        {
            if(grid[i][j]==grid[i][j+1] && grid[i][j]!=0){count++;}
        }
        if(count==2){return true;}
    }
    
    for(j=0;i<3;i++)
    {
        count=0;
        for(i=0;i<=1;j++)
        {
            if(grid[i][j]==grid[i+1][j] && grid[i][j]!=0){count++;}
        }
        if(count==2){return true;}
    }
    
    if(grid[0][0]==grid[1][1] && grid[1][1]==grid[2][2] && grid[0][0]!=0){return true;}
    if(grid[0][2]==grid[1][1] && grid[0][2]==grid[2][0] && grid[0][2]!=0){return true;}
}

void input(int i,vector<vector<int>> &grid)
{
    int p,q;
        if(i%2==0)
        {
            cout<<"Player1 move:\n";
            cin>>p>>q;
            grid[p][q]=5;
        }
        else
        {
            cout<<"Player2 move:\n";
            cin>>p>>q;
            grid[p][q]=-5;
        }
}

int main()
{
    int i,p,q;
    vector<vector<int>> grid(3);
    
    for(i=0;i<3;i++)
    {
        grid[i].resize(3);
    }
    
    i=0;
    while(i<=8)
    {
        input(i,grid);
        
        if(check(grid)){break;}
        i++;
    }
    
    if(i%2==0 && i<=8 ){cout<<"First player wins!\n";}
    else if(i<=8){cout<<"Second player wins!\n";}
    
    if(i==9){cout<<"Tie\n";}
    
    return 0;
}
