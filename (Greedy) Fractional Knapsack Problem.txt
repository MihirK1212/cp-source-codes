#include <iostream>
using namespace std;

int main()
{
    int n,i,j,maxind;
    double W,max=0.0;
    
    cout<<"Enter total capacity of backpack:\n";
    cin>>W;
    
    double left=W;
    
    cout<<"Enter number of items:\n";
    cin>>n;
    
    double w[n],v[n],ans[n];

    
    for(i=0;i<n;i++)
    {
        ans[i]=0;
    }
    
   
    
    cout<<"Enter weight and value of each item\n";
    for(i=0;i<n;i++)
    {
        cout<<"Weight=";
        cin>>w[i];
        cout<<"Value=";
        cin>>v[i];
        cout<<"\n";
    }
    
    
   cout<<"left="<<left<<"\n";
    
    while(left!=0)
    {
        for(i=0;i<n;i++)
      {
       if((v[i]/w[i])>max)
       {
           max=v[i]/w[i];
           maxind=i;
       }
     }
      
      
    
        if(w[maxind]>=left)
        {
            ans[maxind]=left/w[maxind];
            left=0;
        }
        else if(w[maxind]<left)
        {
            left=left-w[maxind];
            ans[maxind]++;
            v[maxind]=0;
        }
        cout<<"left="<<left<<"\n";
        max=0;
        
    }
    
    for(i=0;i<n;i++)
    {
        cout<<"Fraction of item "<<i+1<<" taken is: "<<ans[i]<<"\n";
        
    }
    
    
    
    
    return 0;
    
}
