class Solution{
    public:
        //Function to merge the arrays.
        void merge(long long arr1[], long long arr2[], int n, int m) 
        { 
            int p1=-1,p2=-1;
    
            while((p1+1)<n && (p2+1)<m)
            {
                if(arr1[p1+1]<=arr2[p2+1]){p1++;}
                else{p2++;}
                
                if((p1+p2+2)==n){break;}
            }
            
            while((p1+1)<n && (p1+p2+2)<n){p1++;}
            while((p2+1)<m && (p1+p2+2)<n){p2++;}
            
            // cout<<p1<<" "<<p2<<"\n";
            
            int ind1=p1+1,ind2=0;
            
            while(ind1<n && ind2<=p2)
            {
                int temp = arr1[ind1];
                arr1[ind1]= arr2[ind2];
                arr2[ind2] = temp;
                ind1++; ind2++;
            }
            
            sort(arr1,arr1+n);
            sort(arr2,arr2+m);
        } 
};
