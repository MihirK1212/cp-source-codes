#define sort(a) sort(a.begin(),a.end());
#define rsort(a) sort(a.rbegin(),a.rend());

ll grid_paths(ll m,ll n)
{
    if(m==1 || n==1){return 1;}
    
    return grid_paths(m,n-1)+grid_paths(m-1,n);
}

int main()
{
    ios_base::sync_with_stdio(false); 
    cin.tie(NULL);
    
    ll m,n;
    cin>>m>>n;
    cout<<grid_paths(m,n)<<"\n";
    
    return 0;
}
