#include<iostream>
#include<vector>
using namespace std;

int main()
{
    int t,ans[t],p,n[t],len,k=0,groups=0,i,j,maxpos,temp;
    vector<vector<int>> e;
    cin>>t;
    e.resize(t);
    for(p=0;p<t;p++)
    {
      cin>>n[p];
      e[p].resize(n[p]);
       
    
    for(i=0;i<n[p];i++)
    {
        cin>>e[p][i];
    }
    
    }
    
    for(p=0;p<t;p++)
    {
    groups=0;
    k=0;
    len=n[p];
     for(i=0;i<=len-2;i++)
     {
        maxpos=i;
        for(j=i+1;j<len;j++)
        {
            if(e[p][j]>e[p][maxpos])
            {
                maxpos=j;
            }
            
        }
        temp=e[p][i];
        e[p][i]=e[p][maxpos];
        e[p][maxpos]=temp;
    }
    
    
    
    while(k<=len-1)
    {
      if((k+e[p][k]-1)<=len-1)
     {
        k=k+e[p][k];
        groups++;    
     }
     else
     {
         break;
     }
    
    }
    
    ans[p]=groups;
    }
    
    for(p=0;p<t;p++)
    {
        cout<<ans[p]<<"\n";
    }
     
    
    
    return 0;
}
