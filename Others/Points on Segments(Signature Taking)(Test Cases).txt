#include <iostream>
using namespace std;
int main()
{
    int t,p,n,freq,i;
    int num[t],ans[t];
    int a[t][n],b[t][n];
    cout<<"Enter number of test cases:\n";
    cin>>t;
    for(p=0;p<t;p++)
    {
     cout<<"For test case number "<<p+1<<"\n";
     cout<<"Enter number of segments:\n";
     cin>>num[p];
     n=num[p];
     cout<<"Enter start and end points of "<<n<<" segments\n";
     for(i=0;i<n;i++)
     {
         cin>>a[p][i];
         cin>>b[p][i];
     }
     freq=1;
     for(i=0;i<n;i++)
     {
        if(i!=n-1 && (a[p][i+1]-b[p][i])<=0 )
        {
            continue;
        }
         
         if(i!=n-1 && (a[p][i+1]-b[p][i])>0)
         {
             freq++;
             continue;
         }
         if(i==n-1)
         {
             break;
         }
         
     }
    ans[p]=freq;
    }
    for(p=0;p<t;p++)
    {
        cout<<"Number of points for test case "<<p+1<<" is "<<ans[p]<<"\n";
    }
}
