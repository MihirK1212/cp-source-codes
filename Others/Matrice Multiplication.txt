
#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long m,n,p,q,i,j,k;
    
    cin>>m>>n;
    vector<vector<int>> m1(m);
    for(i=0;i<m;i++){m1[i].resize(n);}
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++){cin>>m1[i][j];}
    }
    
    cout<<"\n";
    
    cin>>p>>q;
    vector<vector<long long>> m2(p);
    for(i=0;i<p;i++){m2[i].resize(q);}
    for(i=0;i<p;i++)
    {
        for(j=0;j<q;j++){cin>>m2[i][j];}
    }
    
    cout<<"\n";
    
    long long sum=0;
   
    if(n!=p)
    {
        cout<<"NOT COMPATIBLE\n";
    }
    
    else
    {
        for(i=0;i<m;i++)
        {
            for(j=0;j<q;j++)
            {
                sum=0;
                for(k=0;k<n;k++)
                {
                    sum+=m1[i][k]*m2[k][j];
                }
                cout<<sum<<" ";
            }
            cout<<"\n";
        }
    }
    
    return 0;
}
