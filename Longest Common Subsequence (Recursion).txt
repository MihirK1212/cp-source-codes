int max_len=-1;
string lcs="";

void solve(string &str1,string &str2,string &comp_str,int index1)
{
    int index_comp=0,index2=0;
    while(index_comp!=comp_str.size() && index2!=str2.size())
    {
        if(comp_str[index_comp]==str2[index2]){index_comp++;index2++;}
        else
        {
            index2++;
        }
    }
    
    if(index_comp==comp_str.size())
    {
        ll len_comp=comp_str.length();
        if(len_comp>max_len){max_len=len_comp; lcs=comp_str;}
    }
    
    if((index1)!=str1.size()){comp_str.push_back(str1[index1]);}
    else{return;}
    solve(str1,str2,comp_str,index1+1);
    comp_str.pop_back();
    solve(str1,str2,comp_str,index1+1);
}
