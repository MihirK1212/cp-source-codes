#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int M,X,Y;
    cin>>X>>Y>>M;
    
    int m,i,j;
    int max=-10;
    
    for(m=M;m>=1;m--)
    {
        for(i=0;i<=(m/X);i++)
        {
            for(j=0;j<=(m/Y);j++)
            {
                if((i*X)+(j*Y)==m)
                {
                    max=m;
                    break;
                }
            }
            if(max!=-10){break;}
        }
        if(max!=-10){break;}
    }
    
    cout<<max<<"\n";
    

    return 0;
}
