#include <iostream>
using namespace std;

int main()
{
    int n,i,t;
    cin>>n;
    int a[n];
    int prev[n],next[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    
    for(i=0;i<n;i++)
    {
        cin>>next[i];
    }
    
    for(t=1;t<=3;t++)
    {
        for(i=0;i<n;i++)
        {
            prev[i]=next[a[i]-1];
        }
        
        for(i=0;i<n;i++)
        {
            next[i]=prev[i];
        }
    }
    
    for(i=0;i<n;i++)
    {
        cout<<next[i]<<"\n";
    }

    return 0;
}
