#include <iostream>
#include <vector>
using namespace std;

struct shell
{
  vector <int> arr;
  int points=0;
  
  int swap(vector<int> &arr,int i1,int i2)
  {
    int temp;
    temp=arr[i1];
    arr[i1]=arr[i2];
    arr[i2]=temp;
    if(arr[i1]==1){return i1;}
    else if(arr[i2]==1){return i2;}
    else{return -10;}
  }

  void pointcalc(vector<int> &arr,int a,int b,int g)
  {
    int x=swap(arr,a-1,b-1);
    if((x+1)==g)
    {
        points++;
    }
    else if (x==-10 && a!=g && b!=g)
    {
        points++;
    }
  }
};



int main()
{
    shell s1,s2,s3;
    
    (s1.arr).resize(3);
    (s2.arr).resize(3);
    (s3.arr).resize(3);
    
    s1.arr[0]=1;
    s1.arr[1]=0;
    s1.arr[2]=0;
    s2.arr[0]=0;
    s2.arr[1]=1;
    s2.arr[2]=0;
    s3.arr[0]=0;
    s3.arr[1]=0;
    s3.arr[2]=1;
    
    int N,i;
    cin>>N;
    int a,b,g;
    for(i=1;i<=N;i++)
    {
        cin>>a>>b>>g;
        s1.pointcalc(s1.arr,a,b,g);
        s2.pointcalc(s2.arr,a,b,g);
        s3.pointcalc(s3.arr,a,b,g);
    }
    
    cout<<max(s1.points,max(s2.points,s3.points));
}
