#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int m,n,k,i,j,p,q;
    cin>>m>>n>>k;
    
    vector<vector<int>> arr(m);
    for(i=0;i<m;i++)
    {
        arr[i].resize(n);
    }
    
    vector<vector<int>> ans(k*m);
    for(i=0;i<k*m;i++)
    {
        ans[i].resize(k*n);
    }
    
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
        {
            cin>>arr[i][j];
        }
    }
    
    int I=0;
    
    
    for(i=0;i<m;i++)
    {
        for(p=1;p<=k;p++)
        {
            int J=0;
            for(j=0;j<n;j++)
            {
                for(q=1;q<=k;q++)
                {
                    ans[I][J]=arr[i][j];
                    J++;
                }
                
            }
            I++;
        }
    }
    
    for(i=0;i<k*m;i++)
    {
        for(j=0;j<k*n;j++)
        {
            cout<<ans[i][j];
        }
        cout<<"\n";
    }
   
    
    return 0;
}
