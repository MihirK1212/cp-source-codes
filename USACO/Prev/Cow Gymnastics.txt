#include <iostream>
using namespace std;

int main()
{
    int k,n,i,cow,j,m;
    cin>>k>>n;
    int arr[n][k];
    
    for(i=0;i<k;i++)
    {
        for(j=0;j<n;j++)
        {
            cin>>cow;
            arr[cow-1][i]=j+1;
        }
    }
    
    int count=0;
    
    for(i=0;i<=n-2;i++)
    {
      for(j=i+1;j<n;j++)
        {
            int higher=0;
            for(m=0;m<k;m++)
            {
                if(arr[i][m]>arr[j][m]){higher++;}
            }
            if(higher==0 || higher==k){count++;}
        }
        
    }
    
    cout<<count<<"\n";
    
    
    
    return 0;
}
