#include <iostream>
using namespace std;

int main()
{
    int N,M;
    cin>>N>>M;
    
    int roaddist[N+1],roadspeed[N+1],cowdist[M+1],cowspeed[M+1];
    
    roaddist[0]=0;
    roadspeed[0]=0;
    cowdist[0]=0;
    cowspeed[0]=0;
    
    int i,x;
    
    for(i=1;i<=N;i++)
    {
        cin>>x>>roadspeed[i];
        roaddist[i]=roaddist[i-1]+x;
    }
    
    for(i=1;i<=M;i++)
    {
        cin>>x>>cowspeed[i];
        cowdist[i]=cowdist[i-1]+x;
    }
    
    int segmentroad=1,segmentcow=1,max=-10;
    
    while(segmentroad<=N && segmentcow<=M)
    {
       int violation=cowspeed[segmentcow]-roadspeed[segmentroad];
       if(violation>max){max=violation;}
         
         if(roaddist[segmentroad]<cowdist[segmentcow]){segmentroad++;}
         else if(roaddist[segmentroad]>cowdist[segmentcow]){segmentcow++;}
         else
         {
             segmentroad++;
             segmentcow++;
         }
    }
    
    cout<<max<<"\n";

    return 0;
}
