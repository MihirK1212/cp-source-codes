#include <bits/stdc++.h>
using namespace std;

#define ll long long 

int main()
{
    ll i;
    string A,B;
    cin>>A;
    cin>>B;
    ll N=A.length();
    bool check=false;
    ll count=0;
    for(i=0;i<N;i++)
    {
        if(A[i]==B[i] && check)
        {
            count++;
            check=false;
        }
        else
        {
            check=true;
        }
    }
    
    cout<<count-1<<"\n";
    
    return 0;
}
